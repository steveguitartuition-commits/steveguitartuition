<script>
const notes4ths = ["C","F","Bb","Eb","Ab","Db","Gb","Cb","E","A","D","G"];
const notes5ths = ["C","G","D","A","E","B","F#","C#","G#","D#","A#","F"];
const audioMap = {
  "C":"C.wav","C#":"Cs_Db.wav","Db":"Cs_Db.wav","D":"D.wav",
  "D#":"Ds_Eb.wav","Eb":"Ds_Eb.wav","E":"E.wav","F":"F.wav",
  "F#":"Fs_Gb.wav","Gb":"Fs_Gb.wav","G":"G.wav","G#":"Gs_Ab.wav",
  "Ab":"Gs_Ab.wav","A":"A.wav","A#":"As_Bb.wav","Bb":"As_Bb.wav",
  "B":"B_Cb.wav","Cb":"B_Cb.wav"
};

const audioElements = {};
for(let n in audioMap){
  const a = new Audio(audioMap[n]);
  a.load();
  audioElements[n] = a;
}
const clickSound = new Audio("button-3.wav");
clickSound.load();

const bpmWaitTimes = {
  50:4800,60:4100,70:3500,80:3200,90:2800,100:2600,
  110:2400,120:2200,130:2100,140:2000,150:1900,160:1800
};

let index4=0,index5=0;
let autoRunning4=false,autoRunning5=false;

function showNote(type){
  document.getElementById(type==='fourths'?'note4ths':'note5ths').textContent =
    type==='fourths'?notes4ths[index4]:notes5ths[index5];
}
function playNote(type){
  const note = type==='fourths'?notes4ths[index4]:notes5ths[index5];
  audioElements[note].currentTime=0;
  audioElements[note].play().catch(()=>{});
}
function stepForward(type){
  if(type==='fourths'){
    index4=(index4+1)%notes4ths.length;
    showNote('fourths'); playNote('fourths');
  } else {
    index5=(index5+1)%notes5ths.length;
    showNote('fifths'); playNote('fifths');
  }
}
function stepBackward(type){
  if(type==='fourths'){
    index4=(index4-1+notes4ths.length)%notes4ths.length;
    showNote('fourths'); playNote('fourths');
  } else {
    index5=(index5-1+notes5ths.length)%notes5ths.length;
    showNote('fifths'); playNote('fifths');
  }
}

function startAuto(type){
  if(type==='fourths'){
    autoRunning4 = true;
    index4 = 0;
  } else {
    autoRunning5 = true;
    index5 = 0;
  }
  startCountIn(type, 1);
}

function stopAuto(type){
  if(type==='fourths'){
    autoRunning4 = false;
    index4 = 0;
    document.getElementById('countdown4ths').textContent = '';
    document.getElementById('note4ths').textContent = 'C';
  } else {
    autoRunning5 = false;
    index5 = 0;
    document.getElementById('countdown5ths').textContent = '';
    document.getElementById('note5ths').textContent = 'C';
  }
}

function startCountIn(type, count){
  if((type==='fourths' && !autoRunning4) || (type==='fifths' && !autoRunning5)) return;

  const countdownDiv = type==='fourths'
    ? document.getElementById('countdown4ths')
    : document.getElementById('countdown5ths');

  const bpm = parseInt(document.getElementById('bpmSelect').value);
  const interval = bpmWaitTimes[bpm] / 4;

  countdownDiv.textContent = count;
  clickSound.currentTime = 0;
  clickSound.play().catch(()=>{});

  if(count < 4) {
    setTimeout(()=> startCountIn(type, count + 1), interval);
  } else {
    setTimeout(()=> startNoteSequence(type), interval);
  }
}

function startNoteSequence(type){
  if((type==='fourths' && !autoRunning4) || (type==='fifths' && !autoRunning5)) return;

  const countdownDiv = type==='fourths'
    ? document.getElementById('countdown4ths')
    : document.getElementById('countdown5ths');

  const bpm = parseInt(document.getElementById('bpmSelect').value);
  const interval = bpmWaitTimes[bpm] / 4;
  let count = 1;

  const incIndex = ()=>{
    if(type==='fourths') index4 = (index4 + 1) % notes4ths.length;
    else index5 = (index5 + 1) % notes5ths.length;
  };

  countdownDiv.textContent = '1';
  showNote(type);
  playNote(type);

  const seqTimer = setInterval(()=>{
    if((type==='fourths' && !autoRunning4) || (type==='fifths' && !autoRunning5)) {
      clearInterval(seqTimer);
      return;
    }
    count++;
    if(count > 4){
      clearInterval(seqTimer);
      incIndex();
      startNoteSequence(type);
    } else {
      countdownDiv.textContent = count;
      if(count > 1){
        clickSound.currentTime = 0;
        clickSound.play().catch(()=>{});
      }
    }
  }, interval);
}
